TypeScript is the future of JavaScript

class Person {

  public name: string;
  public surname: string;

  constructor(name : string, surname : string) {
    this.name = name;
    this.surname = surname;
  }

  public greet() {
    return `Hello! my name is ${this.name} ${this.surname}`;
  }

}

var luke = new Person("Luke", "Skywalker");
console.log(luke.greet());
console.log(luke.surname);

Why to use TypeScript?

TypeScript makes your team more productive

interface IPerson {
  name : string;
  surname : string;
  greet() : string;
}

describe("Person Class", () => {
  it("should be able to greet", () => {
    var luke : IPerson = new Person("Luke", "Skywalker");
    expect(luke.name).to.equal("Luke");
    expect(luke.surname).to.equal("Skywalker");
    expect(luke.greet()).to.equal("Hello! my name is Luke Skywalker");
  });
}
